\newcommand{\mytitle}[1]{
    \node[fill=gray,
        rounded corners,
        draw=white,
        line width=2pt,
        drop shadow,
        text width=.15\textwidth,
        inner sep=8pt,
        xshift=-4cm]
    at (frame.north){\bfseries\textcolor{white}{#1}};
}

\newtcolorbox{mybox}[2][]{
    enhanced,
    enlarge top by=4ex,
    enlarge bottom by=4ex,
    overlay={\mytitle{#2}},
    borderline={2pt}{0mm}{gray},
    borderline={.7pt}{1mm}{gray},
    frame hidden,
    arc=3mm,
    sidebyside,
    lefthand width=.9\textwidth,
    segmentation hidden,
    top=4ex,
    #1
}
